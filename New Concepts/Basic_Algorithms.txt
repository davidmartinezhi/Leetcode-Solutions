Binary search: Iteratively

The point is to keep reducing in half the array or string until we find the element we are looking for

Iterative BS has complexity:
    time: O(log n)
    space: O(1)

int binarySearch(vector<int> & arr, int left, int right, int target){

    //search
    while(left <= right){

        //get middle value index
        int m = left + (right + 1) / 2;

        //check if present at middle
        if(arr[m] == target) return m; 
        else if(arr[m] < target) left  = m+1;
        else right = m - 1;

    }

    //not found
    return -1;

}
